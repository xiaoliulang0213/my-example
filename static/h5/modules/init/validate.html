数据校验(validate属性)

必填
{required: true}
{required: 'xxx不能为空'}
限制为数值
{custom: 'number'}
{custom: {code: 'number', desc: '限制为数字'}}
大小限制
{min: 100, max: 200}
{min: {code: 100, desc: '值太小了'}}
长度限制（应使用maxlength代替）
{minSize: 4}
{minSize: {code: 4, desc: '太短了'}}
日期格式判断
{custom: 'date'}
{custom: {code: 'date', desc: '格式错了'}}
{custom: 'date', future: 'NOW'}
{custom: 'date', future: {code: 'NOW', desc: '日期应晚于今天'}}
{custom: 'date', past: '2016-12-01', future: '2016-08-10'}
邮箱格式
{custom: 'email'}
{custom: {code: 'email', desc: '邮箱格式错了'}}
